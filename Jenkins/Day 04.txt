Step 4: Create Pipeline Job
Repo URL: https://github.com/KastroVKiran/Hotstar-App.git

Script 1:
pipeline {
    agent {
        label 'slave1'
    }

    stages {
        stage('Git Checkout') {
            steps {
                git 'https://github.com/KastroVKiran/Hotstar-App.git'
            }
        }
        stage('Maven Compile') {
            steps {
                sh 'mvn compile'
            }
        }
        stage('Maven Test') {
            steps {
                sh 'mvn test'
            }
        }
        stage('Maven Package') {
            steps {
                sh 'mvn package'
            }
        }
    }
}

Build the job and verify the job information on the specified worker node in /tmp path
